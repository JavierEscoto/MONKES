%%  COMMANDS TO CONSTRUCT GRAPHS

\newcommand{\functiongraph}[7]
{
	\begin{tikzpicture}[]
		\begin{axis}[ title style={at={(0.5,-0.4)},anchor=south}, title={#7}, xlabel={#1}, ylabel={#2}, legend style={font=\small}, cycle list/Set1, width = 0.6\textwidth, {#3}]
			\foreach \f in {#4}
			\addplot {\f};
			{#6}
			\legend{#5}
		\end{axis}
	\end{tikzpicture}
}%

\newcommand{\graph}[7]
{
	\begin{tikzpicture}[]
		\begin{axis}[ width = \textwidth, title style={at={(0.5,-0.4)},anchor=south}, title={#7}, xlabel={#1}, ylabel={#2}, legend style={font=\small}, cycle list/Set1, {#3} ]
			\foreach \x in {#6}
			\addplot table [skip first n=1, x index=0,y index=\x]{#4};
			\legend{#5}
		\end{axis}
	\end{tikzpicture}
}%

\newcommand{\curveplot}[7]
{
	\addplot3[ domain = #1, domain y = #2, restrict z to domain* = #3,
	#7]({ #4 },{ #5 },{ #6 });
}

\newcommand{\vectorplot}[9]
{
	\addplot3[ domain = #1, domain y = #2, domain z = #3, #9
	quiver={
		u={ #4 },
		v={ #5 } ,
		w={ #6 } ,
		scale arrows= #7
	},samples=#8, -stealth]{0};
}

\newcommand{\contour}[6]
{
	\begin{tikzpicture}[]
		\begin{axis}[ height = 0.8\textwidth, width = 0.8\textwidth, title style={at={(0.5,-0.4)}, yshift=-0.1}, title={#6}, view={0}{90}, #1, colorbar]
			\addplot3[smooth,surf,mesh/rows=#2, shader =interp,z buffer=sort,#4]
			%    \addplot3[ contour gnuplot,mesh/rows=#2, shader =interp,z buffer=sort,#4]
			table [skip first n=2] {#3}; 
			{#5}
		\end{axis}
	\end{tikzpicture}
}




\newcommand{\functioncontour}[5]
{
	\begin{tikzpicture}[]
		\begin{axis}[ view={0}{90}, #1 , height = 0.875\textwidth, width = \textwidth, title style={at={(0.5,-0.4)}, yshift=-0.1}, title={#5}]
			\addplot3[{#2}]{#3}; 
			{#4}
		\end{axis}
	\end{tikzpicture}
}
